// client/src/components/home/FieldToCash.tsx
import * as React from "react";
import { flowSteps } from "../../data/fieldToCash";
import { Link } from "react-router-dom";
import { motion, AnimatePresence } from "framer-motion";
import { Calendar, FileText, CreditCard, RefreshCcw } from "lucide-react";

function IconFor({ name }: { name: "calendar" | "file" | "credit" | "repeat" }) {
  switch (name) {
    case "calendar": return <Calendar className="size-5" aria-hidden />;
    case "file": return <FileText className="size-5" aria-hidden />;
    case "credit": return <CreditCard className="size-5" aria-hidden />;
    case "repeat": return <RefreshCcw className="size-5" aria-hidden />;
  }
}

export default function FieldToCash() {
  const [active, setActive] = React.useState(0);
  const prefersReduce =
    typeof window !== "undefined" &&
    window.matchMedia?.("(prefers-reduced-motion: reduce)").matches;

  // Observe which step is most visible
  const refs = React.useMemo(() => flowSteps.map(() => React.createRef<HTMLDivElement>()), []);
  React.useEffect(() => {
    const io = new IntersectionObserver((entries) => {
      let max = { idx: active, ratio: 0 };
      for (const e of entries) {
        const idx = refs.findIndex(r => r.current === e.target);
        if (e.intersectionRatio > max.ratio) max = { idx, ratio: e.intersectionRatio };
      }
      if (max.ratio >= 0.5 && max.idx !== active) setActive(max.idx);
    }, { threshold: [0, 0.5, 1] });
    refs.forEach(r => r.current && io.observe(r.current!));
    return () => io.disconnect();
  }, [refs, active]);

  return (
    <section className="relative isolate py-14 sm:py-16 lg:py-24">
      <div className="container mx-auto px-4 sm:px-6 lg:px-8">
        <header className="max-w-2xl">
          <p className="text-xs uppercase tracking-widest text-white/60">Workflow</p>
          <h2 className="mt-2 text-2xl sm:text-3xl font-semibold text-white">
            From first call to cash—one clean flow
          </h2>
          <p className="mt-3 text-white/80">
            LinkServicePro unifies the everyday steps contractors actually run: schedule the job, win the work, get paid, and lock in recurring revenue.
          </p>
        </header>

        <div className="mt-10 grid gap-8 lg:grid-cols-12 lg:gap-8">
          {/* LEFT: step list */}
          <div className="lg:col-span-7 space-y-5">
            {flowSteps.map((step, i) => (
              <div
                key={step.key}
                ref={refs[i]}
                className={`rounded-xl border bg-white/5 p-4 sm:p-5 transition
                           ${i === active ? "border-white/20 ring-1 ring-white/20 bg-white/7" : "border-white/10"}`}
              >
                <div className="flex items-start gap-3">
                  <span className={`shrink-0 rounded-md p-2 ${i === active ? "bg-white/15" : "bg-white/10"}`}>
                    <IconFor name={step.icon} />
                  </span>
                  <div className="min-w-0">
                    <h3 className="text-lg font-semibold text-white">{step.title}</h3>
                    <p className="mt-1 text-sm text-white/80">{step.blurb}</p>
                    <Link to={step.ctaHref} className="mt-3 inline-flex text-brand-300 hover:text-brand-200 underline underline-offset-4">
                      {step.ctaText} →
                    </Link>
                  </div>
                </div>

                {/* Mobile inline image */}
                <img
                  src={step.img.src}
                  alt={step.img.alt}
                  loading="lazy"
                  decoding="async"
                  className="mt-4 w-full rounded-lg object-cover lg:hidden"
                />
              </div>
            ))}
          </div>

          {/* RIGHT: device card that crossfades on desktop only */}
          <div className="lg:col-span-5 hidden lg:block">
            <div className="sticky top-24">
              <div className="relative h-[56vh] md:h-[60vh] w-full overflow-hidden rounded-xl bg-black/10" aria-hidden>
                <AnimatePresence initial={false}>
                  <motion.img
                    key={prefersReduce ? `static-${active}` : active}
                    src={flowSteps[active].img.src}
                    alt=""
                    className="absolute inset-0 size-full object-cover"
                    initial={{ opacity: prefersReduce ? 1 : 0 }}
                    animate={{ opacity: 1 }}
                    exit={{ opacity: prefersReduce ? 1 : 0 }}
                    transition={{ duration: prefersReduce ? 0 : 0.25, ease: "linear" }}
                    loading="lazy"
                    decoding="async"
                  />
                </AnimatePresence>
                {/* subtle grid overlay for blueprint feel */}
                <div className="absolute inset-0 pointer-events-none bg-[radial-gradient(rgba(255,255,255,0.04)_1px,transparent_1px)] [background-size:16px_16px]" />
              </div>
            </div>
          </div>
        </div>
      </div>
    </section>
  );
}

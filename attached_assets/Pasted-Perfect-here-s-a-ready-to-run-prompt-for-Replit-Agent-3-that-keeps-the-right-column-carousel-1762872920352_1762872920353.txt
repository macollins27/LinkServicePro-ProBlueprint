/** Decorative route path with soft glow. */
function RouteGlow() {
  useEffect(() => {
    const reduce = window.matchMedia("(prefers-reduced-motion: reduce)").matches;
    if (reduce) return;
  }, []);

  return (
    <svg
      aria-hidden="true"
      className="absolute inset-0 w-[120%] h-[120%] -left-10 -top-10 opacity-30"
      viewBox="0 0 1200 800"
      preserveAspectRatio="none"
    >
      <defs>
        <filter id="soft-glow">
          <feGaussianBlur stdDeviation="8" result="blur" />
          <feMerge>
            <feMergeNode in="blur" />
            <feMergeNode in="SourceGraphic" />
          </feMerge>
        </filter>
      </defs>
      <path
        d="M -50 650 C 200 520, 420 720, 620 600 S 980 320, 1250 420"
        fill="none"
        stroke="#2D6BFF"
        strokeOpacity="0.35"
        strokeWidth="2"
        filter="url(#soft-glow)"
        style={{
          strokeDasharray: "8 10",
          animation: "dash-move 14s linear infinite"
        }}
      />
      <style>{`
        @media (prefers-reduced-motion: reduce) {
          svg .route { animation: none !important; }
        }
        @keyframes dash-move { to { stroke-dashoffset: -180; } }
      `}</style>
    </svg>
  );
}

## 2) NAVBAR — Blend on load, solid on scroll

**Goal:** Navbar is transparent over the dark hero at first. Once the user scrolls past the hero, the navbar becomes solid and gains a subtle border.

### Markup/classes

* Initial classes: `bg-transparent text-white`
* When scrolled: add `bg-[#0A1326]/90 backdrop-blur border-b border-white/10`

```tsx
// Navbar.tsx (minimal changes)
import { useEffect, useState } from "react";
export default function Navbar() {
  const [scrolled, setScrolled] = useState(false);

  useEffect(() => {
    const hero = document.querySelector('[aria-label="Hero"]');
    if (!hero) return;

    const obs = new IntersectionObserver(
      ([entry]) => setScrolled(!entry.isIntersecting),
      { threshold: 0.01 }
    );
    obs.observe(hero);
    return () => obs.disconnect();
  }, []);

  return (
    <header
      className={
        "fixed top-0 inset-x-0 z-50 transition-colors " +
        (scrolled
          ? "bg-[#0A1326]/90 backdrop-blur border-b border-white/10 text-white"
          : "bg-transparent text-white")
      }
    >
      {/* existing nav content */}
    </header>
  );
}
```

2. **Update marquee component** – captions + reduced-motion grid

```tsx
// client/src/components/animation/InfiniteVerticalLoop.tsx
import * as React from "react";

type Item = { src: string; alt?: string; label?: string };

export default function InfiniteVerticalLoop({
  items,
  reverse = false,
  speedSeconds = 28,
  className = "",
}: {
  items: Item[];
  reverse?: boolean;
  speedSeconds?: number;
  className?: string;
}) {
  const prefersReduce =
    typeof window !== "undefined" &&
    window.matchMedia?.("(prefers-reduced-motion: reduce)").matches;

  if (prefersReduce) {
    // Static grid with semantic captions
    return (
      <div className={className}>
        <ul className="grid grid-cols-2 gap-3">
          {items.map((it, i) => (
            <li key={i}>
              <figure className="relative">
                <img
                  src={it.src}
                  alt={it.alt ?? ""}
                  loading="lazy"
                  decoding="async"
                  className="w-[140px] sm:w-[160px] aspect-[4/3] object-cover rounded-md"
                />
                {it.label && (
                  <figcaption className="mt-1 text-[11px] sm:text-xs text-white/90">
                    {it.label}
                  </figcaption>
                )}
              </figure>
            </li>
          ))}
        </ul>
      </div>
    );
  }

  // Animated track — duplicate for seamless loop; hide from AT
  const doubled = [...items, ...items];

  return (
    <div
      aria-hidden="true"
      className={`overflow-hidden [mask-image:linear-gradient(to_bottom,transparent,black_12%,black_88%,transparent)] ${className}`}
    >
      <div
        className={`${reverse ? "animate-marqueeDown" : "animate-marqueeUp"} flex flex-col gap-3 will-change-transform`}
        style={{ animationDuration: `${speedSeconds}s` }}
      >
        {doubled.map((it, i) => (
          <figure key={i} className="relative">
            <img
              src={it.src}
              alt="" // decorative while animated
              loading="lazy"
              decoding="async"
              className="w-[140px] sm:w-[160px] aspect-[4/3] object-cover rounded-md"
            />
            {it.label && (
              <figcaption
                className="
                  absolute left-2 bottom-2 z-10
                  rounded px-2 py-1
                  text-[11px] sm:text-xs font-medium
                  text-white bg-black/70 ring-1 ring-white/20
                  pointer-events-none
                "
              >
                {it.label}
              </figcaption>
            )}
          </figure>
        ))}
      </div>
    </div>
  );
}
```

> Assumes Tailwind `marqueeUp`/`marqueeDown` keyframes already exist. If not, add them under `theme.extend` with 28–30s durations.

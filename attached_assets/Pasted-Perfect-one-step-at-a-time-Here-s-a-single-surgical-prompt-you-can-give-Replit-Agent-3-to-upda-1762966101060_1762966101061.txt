**B) Reusable loop**

```tsx
// client/src/components/animation/InfiniteVerticalLoop.tsx
import * as React from "react";

type Item = { src: string; alt?: string; };
export default function InfiniteVerticalLoop({
  items,
  reverse = false,
  speedSeconds = 28,
  className = "",
}: {
  items: Item[];
  reverse?: boolean;
  speedSeconds?: number;
  className?: string;
}) {
  const prefersReduce = window.matchMedia?.("(prefers-reduced-motion: reduce)").matches;

  if (prefersReduce) {
    return (
      <div className={className}>
        <ul className="grid grid-cols-2 gap-3">
          {items.slice(0, 6).map((it, i) => (
            <li key={i}>
              <img
                src={it.src}
                alt={it.alt ?? ""}
                loading="lazy"
                decoding="async"
                className="w-[140px] sm:w-[160px] aspect-[4/3] object-cover rounded-md"
              />
            </li>
          ))}
        </ul>
      </div>
    );
  }

  // duplicate for seamless 200% track
  const doubled = [...items, ...items];

  return (
    <div
      aria-hidden="true"
      className={`overflow-hidden [mask-image:linear-gradient(to_bottom,transparent,black_12%,black_88%,transparent)] ${className}`}
    >
      <div
        className={`${reverse ? "animate-marqueeDown" : "animate-marqueeUp"} flex flex-col gap-3`}
        style={{ animationDuration: `${speedSeconds}s` }}
      >
        {doubled.map((it, i) => (
          <img
            key={i}
            src={it.src}
            alt="" // decorative in motion; descriptive alt retained in reduced-motion path
            loading="lazy"
            decoding="async"
            className="w-[140px] sm:w-[160px] aspect-[4/3] object-cover rounded-md"
          />
        ))}
      </div>
    </div>
  );
}
```

function Waterfall({ bars }) {
  // draw all bars UP from the baseline; labels tucked right under the line
  const W = 520;
  const H = 120;      // smaller canvas = tighter bottom
  const baseY = 80;   // baseline
  const gap = 14;
  const padX = 16;

  const maxAbs = Math.max(1, ...bars.map(b => Math.abs(b.v)));
  const maxBarH = 68;
  const scale = maxBarH / maxAbs;
  const barW = Math.floor((W - padX * 2 - gap * (bars.length - 1)) / bars.length);

  return (
    <div className="w-full">
      <svg
        viewBox={`0 0 ${W} ${H}`}
        width="100%"
        height="120"
        role="img"
        aria-label="Cost breakdown"
        className="block"               // remove inline-SVG baseline gap
      >
        <line x1={padX} y1={baseY} x2={W - padX} y2={baseY} stroke="#E5E7EB" strokeWidth="2" />

        {bars.map((b, i) => {
          const h = Math.max(4, Math.round(Math.abs(b.v) * scale));
          const x = padX + i * (barW + gap);
          const y = baseY - h;          // always above baseline
          return (
            <g key={i}>
              <rect x={x} y={y} width={barW} height={h} rx="2" fill={b.color} />
              <text
                x={x + barW / 2}
                y={baseY + 10}         // tight to baseline
                textAnchor="middle"
                className="fill-slate-700"
                style={{ fontSize: 12 }}
              >
                {b.k}
              </text>
            </g>
          );
        })}
      </svg>
    </div>
  );
}

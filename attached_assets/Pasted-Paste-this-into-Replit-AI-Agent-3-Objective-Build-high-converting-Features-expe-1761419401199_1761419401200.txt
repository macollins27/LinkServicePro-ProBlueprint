## Objective

Build high-converting **Features** experiences that double as landing pages for SEO (e.g., “HVAC company scheduling software”). Ship:

1. A browsable **All Features** showcase with filters, search, and cluster groupings.
2. A **Feature Landing** template used by every feature, optimized for conversion, speed, and SEO.
3. A small **content data model** so new features are one JSON entry.
4. Global lead capture, sticky CTA, and internal-linking patterns.

Mobile-first. Sharp corners. High contrast. One primary action per section. Every section begins with **Title + Subtitle** at **top-left**. Honor `prefers-reduced-motion`.

---

## Global rules

* Framework: keep current stack. If Next.js, use `/pages`; if Vite/CRA, adapt routes 1:1.
* Performance: lazy-load non-critical media; fixed heights for media to avoid CLS; image `loading="lazy"`.
* A11y: semantic landmarks (`header/main/section/nav/footer`), ARIA for accordions, `focus-visible` rings, AA contrast.
* UI system: reuse spacing and `elev-2` shadow. Rectangle buttons with bold labels. Links animate underline right→left.
* No horizontal scroll at 375px. Tap targets ≥44px.

```jsx
import features from "@/data/features.json";
import SectionHeader from "@/components/SectionHeader";
import { useMemo, useState } from "react";

const CLUSTERS = ["Plan & Dispatch","Win & Deliver","Get Paid & Operate"];

function Card({ f }) {
  return (
    <a href={`/features/${f.slug}`} className="bg-white ring-1 ring-slate-200 p-5 sm:p-6 elev-2 block focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-[#0B5FFF]">
      <div className="flex items-start gap-3">
        <span className="h-6 w-6 bg-slate-900" aria-hidden="true" />
        <div>
          <div className="font-extrabold text-slate-900">{f.name}</div>
          <div className="text-slate-700 text-sm">{f.hook}</div>
        </div>
      </div>
      <ul className="mt-3 text-sm text-slate-700 list-disc pl-5 space-y-1">
        {f.bullets.slice(0,2).map((b,i)=><li key={i}>{b}</li>)}
      </ul>
      <div className="mt-4 inline-flex font-semibold text-[#0B5FFF] underline decoration-transparent hover:decoration-[#0B5FFF] underline-offset-4">
        See how it works →
      </div>
    </a>
  );
}

export default function AllFeatures() {
  const [q,setQ]=useState(""); const [cluster,setCluster]=useState("All");
  const filtered = useMemo(()=>{
    return features.filter(f => {
      const matchQ = (f.name+f.hook+f.cluster).toLowerCase().includes(q.toLowerCase());
      const matchC = cluster==="All" || f.cluster===cluster;
      return matchQ && matchC;
    });
  },[q,cluster]);

  return (
    <main className="px-4 sm:px-6 lg:px-8 py-12 sm:py-16">
      <SectionHeader title="All Features" subtitle="Plan, dispatch, deliver, and get paid with one hub." />
      <div className="grid gap-3 sm:flex sm:items-center sm:gap-4">
        <label className="block max-w-xl grow">
          <span className="sr-only">Search features</span>
          <input value={q} onChange={e=>setQ(e.target.value)} placeholder="Search features"
            className="w-full border border-slate-300 px-4 py-3 rounded-none" />
        </label>
        <select value={cluster} onChange={e=>setCluster(e.target.value)}
          className="border border-slate-300 px-3 py-3 rounded-none">
          <option>All</option>
          {CLUSTERS.map(c=><option key={c}>{c}</option>)}
        </select>
      </div>

      {CLUSTERS.map(c=>{
        const group = filtered.filter(f=>f.cluster===c);
        if(!group.length) return null;
        return (
          <section key={c} className="mt-8 sm:mt-10">
            <h3 className="text-lg sm:text-xl font-extrabold text-slate-900">{c}</h3>
            <div className="mt-3 grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 sm:gap-6">
              {group.map(f=><Card key={f.slug} f={f} />)}
            </div>
          </section>
        );
      })}

      <section className="mt-12 sm:mt-16 bg-slate-50 p-6 text-center">
        <div className="text-2xl font-extrabold text-slate-900">Not sure where to start?</div>
        <p className="text-slate-700">See the platform in action.</p>
        <div className="mt-3 flex justify-center gap-3">
          <a href="/pricing" className="px-5 py-3 border border-slate-900 font-extrabold rounded-none">View Pricing</a>
          <a href="/signup" className="px-5 py-3 bg-slate-900 text-white font-extrabold rounded-none">Start Free</a>
        </div>
      </section>
    </main>
  );
}
```

```jsx
import features from "@/data/features.json";
import FeatureLayout from "@/components/FeatureLayout";
import StickyBar from "@/components/StickyBar";
import { HeroBand, WhatItDoes, PainSolve, HowItWorks, SocialProof, WorksWith, Related, DemoAndLead, BottomCTA } from "@/components/FeatureBlocks";

export default function FeaturePage({ f, related }) {
  if(!f) return null;
  return (
    <>

        <HeroBand hook={f.hook} kpis={f.kpis} />

        <HowItWorks items={f.how||[]} />
        <SocialProof quote={f.proof?.quote} person={f.proof?.person} />
        <WorksWith list={f.integrations||[]} />
        <Related items={related} />
        <DemoAndLead bullets={["Live schedule view","Assign crews","Notify customer automatically"]} />
        <BottomCTA />
      </FeatureLayout>
      <StickyBar />
    </>
  );
}

export async function getStaticProps({ params }) {
  const f = features.find(x => x.slug === params.slug) || null;
  const related = (f?.related || []).map(slug => features.find(x => x.slug === slug)).filter(Boolean).slice(0,3);
  return { props: { f, related } };
}

export async function getStaticPaths() {
  return { paths: features.map(f => ({ params: { slug: f.slug } })), fallback: false };
}
```
